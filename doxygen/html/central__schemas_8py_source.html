114@dataclass
115class NameTypeMapping:
116 """A simple dataclass mapping field name to field type."""
117
118 name: str
119 type: str
120
121
122ENTITY_FIELDS: list[NameTypeMapping] = [
123 NameTypeMapping(name="geometry", type="geopoint"),
124 NameTypeMapping(name="project_id", type="string"),
125 NameTypeMapping(name="task_id", type="string"),
126 NameTypeMapping(name="osm_id", type="string"),
127 NameTypeMapping(name="tags", type="string"),
128 NameTypeMapping(name="version", type="string"),
129 NameTypeMapping(name="changeset", type="string"),
130 NameTypeMapping(name="timestamp", type="datetime"),
131 NameTypeMapping(name="status", type="string"),
132]
133
134RESERVED_KEYS = {
135 "name",
136 "label",
137 "uuid",
138} # Add any other reserved keys of odk central in here as needed
139ALLOWED_PROPERTY_PATTERN = re.compile(r"^[a-zA-Z0-9_]+$")
140
141
142def is_valid_property_name(name: str) -> bool:
143 """Check if a property name is valid according to allowed characters pattern."""
144 return bool(ALLOWED_PROPERTY_PATTERN.match(name))
145
146
147def sanitize_key(key: str) -> str:
148 """Rename reserved keys to avoid conflicts with ODK Central's schema."""
149 if key in RESERVED_KEYS:
150 return f"custom_{key}"
151 return key

154def entity_fields_to_list(properties: list[str]) -> list[str]:
155 """Converts a list of Field objects to a list of field names."""
156 sanitized_properties = []
157 for property in properties:
158 if not is_valid_property_name(property):
159 log.warning(f"Invalid property name: {property},Excluding from properties.")
160 continue
161 sanitized_properties.append(sanitize_key(property))
162 default_properties = [field.name for field in ENTITY_FIELDS]
163 for item in default_properties:
164 if item not in sanitized_properties:
165 sanitized_properties.append(item)
166 return sanitized_properties

169# Dynamically generate EntityPropertyDict using ENTITY_FIELDS
170def create_entity_property_dict() -> dict[str, type]:
171 """Dynamically create a TypedDict using the defined fields."""
172 return {field.name: str for field in ENTITY_FIELDS}

275 @field_validator("label", mode="before")
276 @classmethod
277 def append_status_emoji(cls, value: str, info: ValidationInfo) -> str:
278 """Add 🔒 (locked), ✅ (complete) or ❌ (invalid) emojis."""
279 status = info.data.get("status", EntityState.READY.value)
280 emojis = {
281 str(EntityState.OPENED_IN_ODK.value): "🔒",
282 str(EntityState.SURVEY_SUBMITTED.value): "✅",
283 str(EntityState.MARKED_BAD.value): "❌",
284 }
285
286 # Remove any existing emoji at the start of the label
287 for emoji in emojis.values():
288 if value.startswith(emoji):
289 value = value[len(emoji) :].lstrip()
290 break
291
292 if status in emojis.keys():
293 value = f"{emojis[status]} {value}"
294
295 return value
296
