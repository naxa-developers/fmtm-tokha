85def dump_and_check_model(db_model: BaseModel):
86 """Dump the Pydantic model, removing None and default values.
87
88 Also validates to check the model is not empty for insert / update.
89 """
90 model_dump = db_model.model_dump(exclude_none=True, exclude_unset=True)
91
92 if not model_dump:
93 log.error("Attempted create or update with no data.")
94 raise HTTPException(
95 status_code=HTTPStatus.BAD_REQUEST, detail="No data provided."

171 @classmethod
172 async def one(cls, db: Connection, user_identifier: int | str) -> Self:
173 """Get a user either by ID or username, including roles and orgs managed."""
174 async with db.cursor(row_factory=class_row(cls)) as cur:
175 sql = """
176 SELECT
177 u.*,
178
179 -- Aggregate organisation IDs managed by the user
180 array_agg(
181 DISTINCT om.organisation_id
182 ) FILTER (WHERE om.organisation_id IS NOT NULL) AS orgs_managed,
183
184 -- Aggregate project roles for the user, as project:role pairs
185 jsonb_object_agg(
186 ur.project_id,
187 COALESCE(ur.role, 'MAPPER')
188 ) FILTER (WHERE ur.project_id IS NOT NULL) AS project_roles
189
190 FROM users u
191 LEFT JOIN user_roles ur ON u.id = ur.user_id
192 LEFT JOIN organisation_managers om ON u.id = om.user_id
193 """
194
195 if isinstance(user_identifier, int):
196 # Is ID
197 sql += """
198 WHERE u.id = %(user_identifier)s
199 GROUP BY u.id;
200 """
201 else:
202 # Is username (basic flexible matching)
203 sql += """
204 WHERE u.username ILIKE %(user_identifier)s
205 GROUP BY u.id;
206 """
207 user_identifier = f"{user_identifier}%"
208
209 await cur.execute(
210 sql,
211 {"user_identifier": user_identifier},
212 )
213 db_user = await cur.fetchone()
214
215 if db_user is None:
216 raise KeyError(f"User ({user_identifier}) not found.")
217
218 return db_user

792 SELECT
793 the.*,
794 u.profile_img
795 FROM

799 WHERE {filters_joined}
