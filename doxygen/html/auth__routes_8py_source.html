53 """Get Login URL for OSM Oauth Application.
54
55 The application must be registered on openstreetmap.org.

57

59 request: The GET request.

121async def get_or_create_user(
122 db: Connection,
123 user_data: AuthUser,
124):
125 """Get user from User table if exists, else create."""
126 try:
127 upsert_sql = """
128 WITH upserted_user AS (
129 INSERT INTO users (
130 id, username, profile_img, role, registered_at
131 ) VALUES (
132 %(user_id)s, %(username)s, %(profile_img)s, %(role)s, NOW()
133 )
134 ON CONFLICT (id)
135 DO UPDATE SET
136 profile_img = EXCLUDED.profile_img,
137 last_login_at = NOW()
138 RETURNING id, username, profile_img, role
139 )
140
141 SELECT
142 u.id, u.username, u.profile_img, u.role,
143
144 -- Aggregate the organisation IDs managed by the user
145 array_agg(
146 DISTINCT om.organisation_id
147 ) FILTER (WHERE om.organisation_id IS NOT NULL) AS orgs_managed,
148
149 -- Aggregate project roles for the user, as project:role pairs
150 jsonb_object_agg(
151 ur.project_id,
152 COALESCE(ur.role, 'MAPPER')
153 ) FILTER (WHERE ur.project_id IS NOT NULL) AS project_roles
154
155 FROM upserted_user u
156 LEFT JOIN user_roles ur ON u.id = ur.user_id
157 LEFT JOIN organisation_managers om ON u.id = om.user_id
158 GROUP BY u.id, u.username, u.profile_img, u.role;
159 """
160
161 async with db.cursor(row_factory=class_row(DbUser)) as cur:
162 await cur.execute(
163 upsert_sql,
164 {
165 "user_id": user_data.id,
166 "username": user_data.username,
167 "profile_img": user_data.profile_img or "",
168 "role": UserRole(user_data.role).name,
169 },
170 )
171 db_user_details = await cur.fetchone()
172
173 if not db_user_details:
174 raise HTTPException(
175 status_code=HTTPStatus.NOT_FOUND,
176 detail=f"User ID ({user_data.id}) could not be inserted in db",
177 )
178
179 return db_user_details
180
181 except Exception as e:
182 await db.rollback()
183 log.exception(f"Exception occurred: {e}", stack_info=True)
184 if 'duplicate key value violates unique constraint "users_username_key"' in str(
185 e
186 ):
187 raise HTTPException(
188 status_code=HTTPStatus.BAD_REQUEST,
189 detail=f"User with this username {user_data.username} already exists.",
190 ) from e
191 else:
192 raise HTTPException(
193 status_code=HTTPStatus.BAD_REQUEST, detail=str(e)
194 ) from e

235@router.get("/refresh/mapper", response_model=Optional[FMTMUser])
236async def refresh_mapper_token(
237 request: Request,
238 current_user: Annotated[AuthUser, Depends(mapper_login_required)],
239):
240 """Uses the refresh token to generate a new access token.
241
242 This endpoint is specific to the mapper mobile frontend.
243 By default the user will be logged in with a temporary auth cookie.
244 OSM auth is optional, if the user wishes to be attributed for contributions.
245
246 NOTE this endpoint has no db calls and returns in ~2ms.
247 """
248 try:
249 # If standard login cookie is passed, use that
250 response = await refresh_cookies(
251 request,
252 current_user,
253 settings.cookie_name,
254 f"{settings.cookie_name}_refresh",
255 )
256 return response
257 except HTTPException:
258 # NOTE we allow for token verification to fail for the main cookie
259 # and fallback to to generate a temp auth cookie
260 pass
261
262 # Refresh the temp cookies (we must re-create the 'sub' field)
263 temp_jwt_details = {
264 **current_user.model_dump(exclude=["id"]),
265 "sub": f"fmtm|{current_user.id}",
266 "aud": settings.FMTM_DOMAIN,
267 "iat": int(time()),
268 "exp": int(time()) + 86400, # set token expiry to 1 day
269 }
270
271 fmtm_token, refresh_token = create_jwt_tokens(temp_jwt_details)
272 # NOTE be sure to not append content=current_user.model_dump() to this JSONResponse
273 # as we want the login state on the frontend to remain empty (allowing the user to
274 # log in via OSM instead / override)
275 response = JSONResponse(status_code=HTTPStatus.OK, content=temp_jwt_details)
276 return set_cookies(
277 response,
278 fmtm_token,
279 refresh_token,
280 f"{settings.cookie_name}_temp",
281 f"{settings.cookie_name}_temp_refresh",
282 )
