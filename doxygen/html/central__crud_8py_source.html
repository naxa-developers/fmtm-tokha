47def get_odk_project(odk_central: Optional[central_schemas.ODKCentralDecrypted] = None):
48 """Helper function to get the OdkProject with credentials."""
49 if odk_central:
50 url = odk_central.odk_central_url
51 user = odk_central.odk_central_user
52 pw = odk_central.odk_central_password
53 else:
54 log.debug("ODKCentral connection variables not set in function")
55 log.debug("Attempting extraction from environment variables")
56 url = settings.ODK_CENTRAL_URL
57 user = settings.ODK_CENTRAL_USER
58 pw = (
59 settings.ODK_CENTRAL_PASSWD.get_secret_value()
60 if settings.ODK_CENTRAL_PASSWD
61 else ""
62 )
63
64 try:
65 log.debug(f"Connecting to ODKCentral: url={url} user={user}")
66 project = OdkProject(url, user, pw)
67
68 except ValueError as e:
69 log.error(e)
70 raise HTTPException(
71 status_code=HTTPStatus.UNAUTHORIZED,
72 detail=(
73 "ODK credentials are invalid, or may have been updated. "
74 "Please update them."
75 ),
76 ) from e
77 except Exception as e:
78 log.exception(f"Error: {e}", stack_info=True)
79 raise HTTPException(
80 HTTPStatus.INTERNAL_SERVER_ERROR,
81 detail=f"Error creating project on ODK Central: {e}",
82 ) from e
83
84 return project

348async def convert_geojson_to_odk_csv(
349 input_geojson: BytesIO,
350) -> StringIO:
351 """Convert GeoJSON features to ODK CSV format.
352
353 Used for form upload media (dataset) in ODK Central.
354
355 Args:
356 input_geojson (BytesIO): GeoJSON file to convert.
357
358 Returns:
359 feature_csv (StringIO): CSV of features in XLSForm format for ODK.
360 """
361 parsed_geojson = parse_geojson_file_to_featcol(input_geojson.getvalue())
362
363 if not parsed_geojson:
364 raise HTTPException(
365 status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
366 detail="Conversion GeoJSON --> CSV failed",
367 )
368
369 csv_buffer = StringIO()
370 csv_writer = csv.writer(csv_buffer)
371
372 header = ["osm_id", "tags", "version", "changeset", "timestamp", "geometry"]
373 csv_writer.writerow(header)
374
375 features = parsed_geojson.get("features", [])
376 for feature in features:
377 geometry = feature.get("geometry")
378 javarosa_geom = await geojson_to_javarosa_geom(geometry)
379
380 properties = feature.get("properties", {})
381 osm_id = properties.get("osm_id")
382 tags = properties.get("tags")
383 version = properties.get("version")
384 changeset = properties.get("changeset")
385 timestamp = properties.get("timestamp")
386
387 csv_row = [osm_id, tags, version, changeset, timestamp, javarosa_geom]
388 csv_writer.writerow(csv_row)
389
390 # Reset buffer position to start to .read() works
391 csv_buffer.seek(0)
392
393 return csv_buffer

396def flatten_json(data: dict, target: dict):
397 """Flatten json properties to a single level.
398
399 Removes any existing GeoJSON data from captured GPS coordinates in
400 ODK submission.
401
402 Usage:
403 new_dict = {}
404 flatten_json(original_dict, new_dict)
405 """
406 for k, v in data.items():
407 if isinstance(v, dict):
408 if "type" in v and "coordinates" in v:
409 # GeoJSON object found, skip it
410 continue
411 flatten_json(v, target)
412 else:
413 target[k] = v

416async def convert_odk_submission_json_to_geojson(
417 input_json: Union[BytesIO, list],
418) -> geojson.FeatureCollection:
419 """Convert ODK submission JSON file to GeoJSON.
420
421 Used for loading into QGIS.
422
423 Args:
424 input_json (BytesIO): ODK JSON submission list.
425
426 Returns:
427 geojson (BytesIO): GeoJSON format ODK submission.
428 """
429 if isinstance(input_json, list):
430 submission_json = input_json
431 else:
432 submission_json = json.loads(input_json.getvalue())
433
434 if not submission_json:
435 raise HTTPException(
436 status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
437 detail="Project contains no submissions yet",
438 )
439
440 all_features = []
441 for submission in submission_json:
442 # Remove unnecessary keys
443 keys_to_remove = ["meta", "__id", "__system"]
444 for key in keys_to_remove:
445 submission.pop(key)
446
447 # Ensure no nesting of the properties (flat struct)
448 data = {}
449 flatten_json(submission, data)
450
451 # Process primary geometry
452 geojson_geom = await javarosa_to_geojson_geom(data.pop("xlocation", {}))
453
454 # Identify and process additional geometries
455 additional_geometries = []
456 for geom_field in list(data.keys()):
457 if geom_field.endswith("_geom"):
458 id_field = geom_field[:-5] # Remove "_geom" suffix
459 geom_data = data.pop(geom_field, {})
460
461 # Convert geometry
462 geom = await javarosa_to_geojson_geom(geom_data)
463
464 feature = geojson.Feature(
465 id=data.get(id_field),
466 geometry=geom,
467 properties={
468 "is_additional_geom": True,
469 "id_field": id_field,
470 "geom_field": geom_field,
471 },
472 )
473 additional_geometries.append(feature)
474
475 feature = geojson.Feature(
476 id=data.get("xlocation"),
477 geometry=geojson_geom,
478 properties=data,
479 )
480 all_features.append(feature)
481 all_features.extend(additional_geometries)
482
483 return geojson.FeatureCollection(features=all_features)

486async def feature_geojson_to_entity_dict(
487 feature: geojson.Feature,
488 additional_features: bool = False,
489) -> central_schemas.EntityDict:
490 """Convert a single GeoJSON to an Entity dict for upload."""
491 if not isinstance(feature, (dict, geojson.Feature)):
492 log.error(f"Feature not in correct format: {feature}")
493 raise ValueError(f"Feature not in correct format: {type(feature)}")
494
495 geometry = feature.get("geometry", {})
496 if not geometry:
497 msg = "'geometry' data field is mandatory"
498 log.debug(msg)
499 raise ValueError(msg)
500
501 javarosa_geom = await geojson_to_javarosa_geom(geometry)
502 raw_properties = feature.get("properties", {})
503 properties = {
504 central_schemas.sanitize_key(key): str(
505 value
506 ) # NOTE all properties MUST be string values for Entities
507 for key, value in raw_properties.items()
508 if central_schemas.is_valid_property_name(key)
509 }
510 if additional_features:
511 entity_label = f"Additional Feature {uuid4()}"
512 else:
513 properties["status"] = "0"
514 feature_id = properties.get("osm_id", None)
515 entity_label = f"House id {feature_id}"
516
517 return {
518 "label": entity_label,
519 "data": {"geometry": javarosa_geom, **properties},
520 }

523async def task_geojson_dict_to_entity_values(
524 task_geojson_dict: Union[dict[int, geojson.Feature], geojson.FeatureCollection],
525 additional_features: bool = False,
526) -> list[central_schemas.EntityDict]:
527 """Convert a dict of task GeoJSONs into data for ODK Entity upload."""
528 log.debug("Converting dict of task GeoJSONs to Entity upload format")
529
530 asyncio_tasks = []
531
532 if additional_features:
533 features = task_geojson_dict.get("features", [])
534 asyncio_tasks.extend(
535 feature_geojson_to_entity_dict(feature, additional_features)
536 for feature in features
537 if feature
538 )
539 else:
540 for geojson_dict in task_geojson_dict.values():
541 features = geojson_dict.get("features", [])
542 asyncio_tasks.extend(
543 feature_geojson_to_entity_dict(feature)
544 for feature in features
545 if feature
546 )
547
548 return await gather(*asyncio_tasks)

605async def get_entities_geojson(
606 odk_creds: central_schemas.ODKCentralDecrypted,
607 odk_id: int,
608 dataset_name: str = "features",
609 minimal: Optional[bool] = False,
610) -> geojson.FeatureCollection:
611 """Get the Entity details for a dataset / Entity list.
612
613 Uses the OData endpoint from ODK Central.
614
615 Currently it is not possible to filter via OData filters on custom params.
616 TODO in the future filter by task_id via the URL,
617 instead of returning all and filtering.
618
619 Response GeoJSON format:
620 {
621 "type": "FeatureCollection",
622 "features": [
623 "type": "Feature",
624 "geometry": {
625 "type": "Polygon",
626 "coordinates": [xxx]
627 },
628 "id": uuid_of_entity,
629 "properties": {
630 "updated_at": "2024-04-11T18:23:30.787Z",
631 "project_id": "1",
632 "task_id": "1",
633 "osm_id": "2",
634 "tags": "xxx",
635 "version": "1",
636 "changeset": "1",
637 "timestamp": "2024-12-20",
638 "status": "LOCKED_FOR_MAPPING"
639 }
640 ]
641 }
642
643
644 Response GeoJSON format, minimal:
645 {
646 "type": "FeatureCollection",
647 "features": [
648 "type": "Feature",
649 "geometry": {
650 "type": "Polygon",
651 "coordinates": [xxx]
652 },
653 "id": uuid_of_entity,
654 "properties": {
655 "osm_id": "0044554",
656 "updated_at": "2024-04-11T18:23:30.787Z",
657 "status": "LOCKED_FOR_MAPPING"
658 }
659 ]
660 }
661
662 Args:
663 odk_creds (ODKCentralDecrypted): ODK credentials for a project.
664 odk_id (str): The project ID in ODK Central.
665 dataset_name (str): The dataset / Entity list name in ODK Central.
666 minimal (bool): Remove all fields apart from id, updated_at, and status.
667
668 Returns:
669 dict: Entity data in OData JSON format.
670 """
671 async with central_deps.get_odk_dataset(odk_creds) as odk_central:
672 entities = await odk_central.getEntityData(
673 odk_id,
674 dataset_name,
675 url_params="$select=__id, __system/updatedAt, geometry, osm_id, status"
676 if minimal
677 else None,
678 )
679
680 all_features = []
681 for entity in entities:
682 flattened_dict = {}
683 flatten_json(entity, flattened_dict)
684
685 javarosa_geom = flattened_dict.pop("geometry") or ""
686 geojson_geom = await javarosa_to_geojson_geom(javarosa_geom)
687
688 feature = geojson.Feature(
689 geometry=geojson_geom,
690 id=flattened_dict.pop("__id"),
691 properties=flattened_dict,
692 )
693 all_features.append(feature)
694
695 return geojson.FeatureCollection(features=all_features)

698async def get_entities_data(
699 odk_creds: central_schemas.ODKCentralDecrypted,
700 odk_id: int,
701 dataset_name: str = "features",
702 fields: str = "__system/updatedAt, osm_id, status, task_id, submission_ids",
703 filter_date: Optional[datetime] = None,
704) -> list:
705 """Get all the entity mapping statuses.
706
707 No geometries are included.
708
709 Args:
710 odk_creds (ODKCentralDecrypted): ODK credentials for a project.
711 odk_id (str): The project ID in ODK Central.
712 dataset_name (str): The dataset / Entity list name in ODK Central.
713 fields (str): Extra fields to include in $select filter.
714 __id is included by default.
715 filter_date (datetime): Filter entities last updated after this date.
716
717 Returns:
718 list: JSON list containing Entity info. If updated_at is included,
719 the format is string 2022-01-31T23:59:59.999Z.
720 """
721 try:
722 url_params = f"$select=__id{',' if fields else ''} {fields}"
723
724 filters = []
725 if filter_date:
726 filters.append(f"__system/updatedAt gt {filter_date}")
727 if filters:
728 url_params += f"&$filter={' and '.join(filters)}"
