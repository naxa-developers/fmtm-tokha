41 """Return an organisation from the DB, else exception.
42
43 Args:
44 id (str | int): The organisation ID (integer) or name (string) to check.
45 db (Connection): The database connection.
46 check_approved (bool): Throw error if org is not approved yet.
47
48 Returns:
49 DbOrganisation: The organisation if found.
50
51 Raises:
52 HTTPException: Raised with a 404 status code if the user is not found.
53 """
54 try:
55 try:
56 # Is ID (int)
57 id = int(id)
58 except ValueError:
59 # Is name (str)
60 pass
61 db_org = await DbOrganisation.one(db, id)
62
63 except KeyError as e:
64 raise HTTPException(status_code=HTTPStatus.NOT_FOUND, detail=str(e)) from e
65
66 if check_approved and db_org.approved is False:
67 raise HTTPException(
68 status_code=HTTPStatus.FORBIDDEN,
69 detail=f"Organisation ({id}) is not approved yet.",
70 )
71
72 return db_org
