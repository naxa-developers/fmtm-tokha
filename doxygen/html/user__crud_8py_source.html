38async def process_inactive_users(
39 db: Connection,
40):
41 """Identify inactive users, send warnings, and delete accounts."""
42 now = datetime.now(timezone.utc)
43 warning_thresholds = [
44 (now - timedelta(days=INACTIVITY_THRESHOLD - days))
45 for days in WARNING_INTERVALS
46 ]
47 deletion_threshold = now - timedelta(days=INACTIVITY_THRESHOLD)
48
49 async with db.cursor() as cur:
50 # Users eligible for warnings
51 for days, warning_date in zip(
52 WARNING_INTERVALS, warning_thresholds, strict=False
53 ):
54 async with db.cursor(row_factory=class_row(DbUser)) as cur:
55 await cur.execute(
56 """
57 SELECT id, username, last_login_at
58 FROM users
59 WHERE last_login_at < %(warning_date)s
60 AND last_login_at >= %(next_warning_date)s;
61 """,
62 {
63 "warning_date": warning_date,
64 "next_warning_date": warning_date - timedelta(days=7),
65 },
66 )
67 users_to_warn = await cur.fetchall()
68
69 for user in users_to_warn:
70 if SVC_OSM_TOKEN:
71 await send_warning_email_or_osm(
72 user.id, user.username, days, SVC_OSM_TOKEN
73 )
74 else:
75 log.warning(
76 f"The SVC_OSM_TOKEN is not set on this server. "
77 f"Cannot send emails to inactive users: "
78 f"{', '.join(user.username for user in users_to_warn)}"
79 )
80
81 # Users eligible for deletion
82 async with db.cursor(row_factory=class_row(DbUser)) as cur:
83 await cur.execute(
84 """
85 SELECT id, username
86 FROM users
87 WHERE last_login_at < %(deletion_threshold)s;
88 """,
89 {"deletion_threshold": deletion_threshold},
90 )
91 users_to_delete = await cur.fetchall()
92
93 for user in users_to_delete:
94 log.info(f"Deleting user {user.username} due to inactivity.")
95 await DbUser.delete(db, user.id)

98async def send_warning_email_or_osm(
99 user_id: int,
100 username: str,
101 days_remaining: int,
102 osm_token: str,
103):
104 """Send warning email or OSM message to the user."""
105 message_content = dedent(f"""
106 ## Account Deletion Warning
107
108 Hi {username},
109
110 Your account has been inactive for a long time. To comply with our policy, your
111 account will be deleted in {days_remaining} days if you do not log in.
112
113 Please log in to reset your inactivity period and avoid deletion.
114
115 Thank you for being a part of our platform!
116 """)
117
118 send_osm_message(
119 osm_token=osm_token,
120 osm_id=user_id,
121 title="FMTM account deletion warning",
122 body=message_content,
123 )
124 log.info(f"Sent warning to {username}: {days_remaining} days remaining.")

127async def get_paginated_users(
128 db: Connection,
129 page: int,
130 results_per_page: int,
131 search: Optional[str] = None,
132) -> dict:
133 """Helper function to fetch paginated users with optional filters."""
134 # Get subset of users
135 users = await DbUser.all(db, search=search)
136 start_index = (page - 1) * results_per_page
137 end_index = start_index + results_per_page
138 paginated_users = users[start_index:end_index]
139
140 pagination = await get_pagination(
141 page, len(paginated_users), results_per_page, len(users)
142 )
143
144 return {"results": paginated_users, "pagination": pagination}
