105 @field_validator("hashtags", mode="before")
106 @classmethod
107 def validate_hashtags(
108 cls,
109 hashtags: Optional[str | list[str]],
110 ) -> Optional[list[str]]:
111 """Validate hashtags.
112
113 - Receives a string and parsed as a list of tags.
114 - Commas or semicolons are replaced with spaces before splitting.
115 - Add '#' to hashtag if missing.
116 """
117 if hashtags is None:
118 return None
119
120 if isinstance(hashtags, str):
121 hashtags = hashtags.replace(",", " ").replace(";", " ")
122 hashtags_list = hashtags.split()
123 else:
124 hashtags_list = hashtags
125
126 # Add '#' to hashtag strings if missing
127 return [
128 f"#{hashtag}" if hashtag and not hashtag.startswith("#") else hashtag
129 for hashtag in hashtags_list
130 ]

307 @computed_field
308 @property
309 def format(self) -> Optional[str]:
310 """Get the basemap format from file extension."""
311 if not self.urlurl:
312 return None
313 return Path(self.urlurl).suffix[1:]
314
315 @computed_field
316 @property
317 def mimetype(self) -> Optional[str]:
318 """Set the mimetype based on the extension."""
319 if not self.urlurl:
320 return None
321
322 file_format = Path(self.urlurl).suffix[1:]
323 if file_format == "mbtiles":
324 return "application/vnd.mapbox-vector-tile"
325 elif file_format == "pmtiles":
326 return "application/vnd.pmtiles"
327 else:
328 return "application/vnd.sqlite3"
