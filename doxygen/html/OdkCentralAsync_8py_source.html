38class OdkCentral(object):
39 """Helper methods for ODK Central API."""
40
41 def __init__(
42 self,
43 url: Optional[str] = None,
44 user: Optional[str] = None,
45 passwd: Optional[str] = None,
46 ):
47 """A Class for accessing an ODK Central server via it's REST API.
48
49 Args:
50 url (str): The URL of the ODK Central
51 user (str): The user's account name on ODK Central
52 passwd (str): The user's account password on ODK Central
53 session (str): Pass in an existing session for reuse.
54
55 Returns:
56 (OdkCentral): An instance of this class
57 """
58 if not url:
59 url = os.getenv("ODK_CENTRAL_URL", default=None)
60 self.url = url
61 if not user:
62 user = os.getenv("ODK_CENTRAL_USER", default=None)
63 self.user = user
64 if not passwd:
65 passwd = os.getenv("ODK_CENTRAL_PASSWD", default=None)
66 self.passwd = passwd
67 verify = os.getenv("ODK_CENTRAL_SECURE", default=True)
68 if type(verify) == str:
69 self.verify = verify.lower() in ("true", "1", "t")
70 else:
71 self.verify = verify
72
73 # Base URL for the REST API
74 self.version = "v1"
75 self.base = f"{self.url}/{self.version}/"
76
77 def __enter__(self):
78 """Sync context manager not allowed."""
79 raise RuntimeError("Must be called with async context manager 'async with'")
80
81 def __exit__(self):
82 """Sync context manager not allowed."""
83 raise RuntimeError("Must be called with async context manager 'async with'")

85 async def __aenter__(self):
86 """Async object instantiation."""
87 # Header enables persistent connection, creates a cookie for this session
88 self.session = aiohttp.ClientSession(
89 raise_for_status=True,
90 headers={"accept": "odkcentral"},
91 )
92 await self.authenticate()
93 return self
94
95 async def __aexit__(self, exc_type, exc_value, traceback):
96 """Async object close."""
97 if self.session:
98 await self.session.close()

100 async def authenticate(self):
101 """Authenticate to an ODK Central server."""
102 try:
103 async with self.session.post(f"{self.base}sessions", json={"email": self.user, "password": self.passwd}) as response:
104 token = (await response.json())["token"]
105 self.session.headers.update({"Authorization": f"Bearer {token}"})
106 except aiohttp.ClientConnectorError as request_error:
107 await self.session.close()
108 raise ConnectionError("Failed to connect to Central. Is the URL valid?") from request_error
109 except aiohttp.ClientResponseError as response_error:
110 await self.session.close()
111 if response_error.status == 401:
112 raise ConnectionError("ODK credentials are invalid, or may have changed. Please update them.") from response_error
113 raise response_error

184 async def getAllProjectSubmissions(self, projectId: int, xforms: list = None, filters: dict = None):
185 """Fetch a list of submissions in a project on an ODK Central server.
186
187 Args:
188 projectId (int): The ID of the project on ODK Central
189 xforms (list): The list of XForms to get the submissions of
190
191 Returns:
192 (json): All of the submissions for all of the XForm in a project
193 """
194 log.info(f"Getting all submissions for ODK project ({projectId}) forms ({xforms})")
195 submission_data = []
196
197 submission_tasks = [self.listSubmissions(projectId, task, filters) for task in xforms]
198 submissions = await gather(*submission_tasks, return_exceptions=True)
199
200 for submission in submissions:
201 if isinstance(submission, Exception):
202 log.error(f"Failed to get submissions: {submission}")
203 continue
204 log.debug(f"There are {len(submission['value'])} submissions")
205 submission_data.extend(submission["value"])
206
207 return submission_data

260 async def createDataset(
261 self,
262 projectId: int,
263 datasetName: Optional[str] = "features",
264 properties: Optional[list[str]] = [],
265 ):
266 """Creates a dataset for a given project.
267
268 Args:
269 projectId (int): The ID of the project to create the dataset for.
270 datasetName (str): The name of the dataset to be created.
271 properties (list[str]): List of property names to create.
272 Alternatively call createDatasetProperty for each property manually.
273
274 Returns:
275 dict: The JSON response containing information about the created dataset.
276
277 Raises:
278 aiohttp.ClientError: If an error occurs during the dataset creation process.
279 """
280 # Validation of properties param
281 if properties and (not isinstance(properties, list) or not isinstance(properties[-1], str)):
282 msg = "The properties must be a list of string values to create a dataset"
283 log.error(msg)
284 raise ValueError(msg)
285
286 # Create the dataset
287 url = f"{self.base}projects/{projectId}/datasets"
288 payload = {"name": datasetName}
289 try:
290 log.info(f"Creating dataset ({datasetName}) for ODK project ({projectId})")
291 async with self.session.post(
292 url,
293 ssl=self.verify,
294 json=payload,
295 ) as response:
296 if response.status not in (200, 201):
297 error_message = await response.text()
298 log.error(f"Failed to create Dataset: {error_message}")
299 log.info(f"Successfully created Dataset {datasetName}")
300 dataset = await response.json()
301 except aiohttp.ClientError as e:
302 msg = f"Failed to create Dataset: {e}"
303 log.error(msg)
304 raise aiohttp.ClientError(msg) from e
305
306 if not properties:
307 return dataset
308
309 # Add the properties, if specified
310 # FIXME this is a bit of a hack until ODK Central has better support
311 # FIXME for adding dataset properties in bulk
312 try:
313 log.debug(f"Creating properties for dataset ({datasetName}): {properties}")
314 properties_tasks = [self.createDatasetProperty(projectId, field, datasetName) for field in properties]
315 success = await gather(*properties_tasks, return_exceptions=True) # type: ignore
316 if not success:
317 log.warning(f"No properties were uploaded for ODK project ({projectId}) dataset name ({datasetName})")
318 except aiohttp.ClientError as e:
319 msg = f"Failed to create properties: {e}"
320 log.error(msg)
321 raise aiohttp.ClientError(msg) from e
322
323 # Manually append to prevent another API call
324 dataset["properties"] = properties
325 return dataset
