43async def generate_api_token(
44 db: Connection,
45 user_id: int,
46) -> str:
47 """Generate a new API token for a given user."""
48 async with db.cursor(row_factory=class_row(DbUser)) as cur:
49 await cur.execute(
50 """
51 UPDATE users
52 SET api_key = %(api_key)s
53 WHERE id = %(user_id)s
54 RETURNING *;
55 """,
56 {"user_id": user_id, "api_key": token_urlsafe(32)},
57 )
58 db_user = await cur.fetchone()
59 if not db_user.api_key:
60 msg = f"Failed to generate API Key for user ({user_id})"
61 log.error(msg)
62 raise ValueError(msg)
63
64 return db_user.api_key

67async def update_entity_status_in_fmtm(
68 db: Connection,
69 odk_event: OdkCentralWebhookRequest,
70):
71 """Update the status for an Entity in the FMTM db."""
72 log.debug(f"Webhook called with event ({odk_event.type.value})")
73
74 if odk_event.type == OdkWebhookEvents.UPDATE_ENTITY:
75 # insert state into db
76 new_state = odk_event.data.get("status")
77
78 if new_state is None:
79 log.warning(f"Missing entity state in webhook event: {odk_event.data}")
80 return HTTPException(
81 status_code=HTTPStatus.BAD_REQUEST,
82 content="Missing entity state property",
83 )
84
85 try:
86 new_state_int = int(new_state)
87 # the string name is inserted in the db
88 new_entity_state = EntityState(new_state_int).name
89 except (ValueError, TypeError):
90 log.warning(
91 f"Invalid entity state '{new_state}' in webhook event: {odk_event.data}"
92 )
93 return HTTPException(
94 status_code=HTTPStatus.BAD_REQUEST,
95 content="Invalid entity state",
96 )
97
98 log.debug(
99 f"Updating entity ({str(odk_event.id)}) status "
100 f"in FMTM db to ({new_entity_state})"
101 )
102 update_success = await DbOdkEntities.update(
103 db,
104 str(odk_event.id),
105 OdkEntitiesUpdate(status=new_entity_state),
106 )
107 if not update_success:
108 raise HTTPException(
109 status_code=HTTPStatus.INTERNAL_SERVER_ERROR,
110 detail=f"Error updating entity with UUID ({str(odk_event.id)})",
111 )
112 return Response(status_code=HTTPStatus.OK)
