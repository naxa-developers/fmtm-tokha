36async def valid_api_token(
37 db: Annotated[Connection, Depends(db_conn)],
38 x_api_key: Annotated[Optional[str], Header()] = None,
39) -> DbUser:
40 """Check the API token is present for an active database user.
41
42 A header X-API-Key must be provided in the request.
43
44 TODO currently this only checks for a valid key, but does not
45 TODO include checking roles.
46 TODO If roles other than 'mapper' are required, this should be integrated.
47 """
48 if not x_api_key:
49 raise HTTPException(
50 status_code=HTTPStatus.FORBIDDEN,
51 detail=(
52 "An API key must be provided in request headers as: X-API-KEY: xxx"
53 ),
54 )
55
56 async with db.cursor(row_factory=class_row(DbUser)) as cur:
57 await cur.execute(
58 """
59 SELECT *
60 FROM users
61 WHERE api_key = %(api_key)s
62 AND is_email_verified = TRUE;
63 """,
64 {"api_key": x_api_key},
65 )
66 db_user = await cur.fetchone()
67 if not db_user:
68 raise HTTPException(
69 status_code=HTTPStatus.UNAUTHORIZED,
70 detail=f"API key invalid: ({x_api_key})",
71 )
72
73 return db_user
