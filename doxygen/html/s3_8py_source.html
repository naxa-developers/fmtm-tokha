149def get_obj_from_bucket(bucket_name: str, s3_path: str) -> BytesIO:
150 """Download an S3 object from a bucket and return it as a BytesIO object.
151
152 Args:
153 bucket_name (str): The name of the S3 bucket.
154 s3_path (str): The path to the S3 object in the bucket.
155
156 Returns:
157 BytesIO: A BytesIO object containing the content of the downloaded S3 object.
158 """
159 # Ensure s3_path starts with a forward slash
160 if not s3_path.startswith("/"):
161 s3_path = f"/{s3_path}"
162
163 client = s3_client()
164 response = None

170 raise ValueError(str(e)) from e
171 finally:
172 if response:
173 response.close()
174 response.release_conn()

219async def delete_all_objs_under_prefix(bucket_name: str, s3_path: str) -> bool:
220 """Delete all objects under a certain path.
221
222 For example, delete all items in a project folder.
223
224 Args:
225 bucket_name (str): The name of the S3 bucket.
226 s3_path (str): The path to for which everything will be deleted recursively.
227
228 Returns:
229 bool: True if the objects were successfully deleted.
230 """
231 # Strip "/" from start of s3_path (not required for delete object)
232 if s3_path.startswith("/"):
233 s3_path = s3_path.lstrip("/")
234 # However, we should append a tailing slash
235 if not s3_path.endswith("/"):
236 s3_path = f"{s3_path}/"
237
238 client = s3_client()
239
240 try:
241 log.info(f"Deleting all items for bucket ({bucket_name}) under path: {s3_path}")
242 # Get all objects under path, recursively
243 objects = list(client.list_objects(bucket_name, s3_path, recursive=True))
244 delete_object_list = [DeleteObject(obj.object_name) for obj in objects]
245 log.debug(f"S3 items to delete: {[obj.object_name for obj in objects]}")
246
247 errors = client.remove_objects(bucket_name, delete_object_list)
248 for error in errors:
249 log.error(f"Error during deletion: {error}")
250 if errors:
251 return False
252
253 except Exception as e:
254 log.exception(
255 f"Failed to delete bucket ({bucket_name}) files under path: {s3_path}. "
256 f"Error: {e}",
257 stack_info=True,
258 )
259 return False
260
261 log.debug("Successfully deleted S3 objects")
262 return True

294def is_connection_secure(minio_url: str):
295 """Determine from URL string if is http or https."""
296 if minio_url.startswith("http://"):
297 secure = False
298 stripped_url = minio_url[len("http://") :]
299 log.warning("S3 URL is insecure (ignore if on devserver)")
300
301 elif minio_url.startswith("https://"):
302 secure = True
303 stripped_url = minio_url[len("https://") :]
304
305 else:
306 err = (
307 "The S3_ENDPOINT is set incorrectly. It must start with http:// or https://"
308 )
309 log.error(err)
310 raise ValueError(err)
311
312 return stripped_url, secure
