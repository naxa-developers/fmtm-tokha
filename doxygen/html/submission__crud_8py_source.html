199async def create_new_submission(
200 odk_credentials: central_schemas.ODKCentralDecrypted,
201 odk_project_id: int,
202 odk_form_id: uuid.UUID,
203 submission_xml: str,
204 device_id: Optional[str] = None,
205 submission_attachments: Optional[dict[str, BytesIO]] = None,
206):
207 """Create a new submission in ODK Central, using pyodk REST endpoint."""
208 submission_attachments = submission_attachments or {} # Ensure always a dict
209 attachment_filepaths = []
210
211 # Write all uploaded data to temp files for upload (required by PyODK)
212 # We must use TemporaryDir and preserve the uploaded file names
213 with TemporaryDirectory() as temp_dir:
214 for file_name, file_data in submission_attachments.items():
215 temp_path = Path(temp_dir) / file_name
216 with open(temp_path, "wb") as temp_file:
217 temp_file.write(file_data.getvalue())
218 attachment_filepaths.append(temp_path)
219
220 pyodk_config = CentralConfig(
221 base_url=odk_credentials.odk_central_url,
222 username=odk_credentials.odk_central_user,
223 password=odk_credentials.odk_central_password,
224 )
225 with Client(pyodk_config) as client:
226 return client.submissions.create(
227 project_id=odk_project_id,
228 form_id=odk_form_id,
229 xml=submission_xml,
230 device_id=device_id,
231 attachments=attachment_filepaths,
232 )

239 """Get the details of a submission.
240

250 project.odkid, project.odk_form_id, submission_id
