80@router.post("/append-geojson-properties")
81async def append_required_geojson_properties(
82 geojson: UploadFile,
83 current_user: Annotated[AuthUser, Depends(login_required)],
84):
85 """Append required properties to a GeoJSON file.
86
87 The required properties for FMTM are:
88 - "id"
89 - "osm_id"
90 - "tags"
91 - "version"
92 - "changeset"
93 - "timestamp"
94
95 These are added automatically if missing during the project creation workflow.
96 However it may be useful to run your file through this endpoint to validation.
97 """
98 featcol = parse_geojson_file_to_featcol(await geojson.read())
99 if not featcol:
100 raise HTTPException(
101 status_code=HTTPStatus.UNPROCESSABLE_ENTITY, detail="No geometries present"
102 )
103 featcol_single_geom_type = featcol_keep_single_geom_type(featcol)
104
105 if featcol_single_geom_type:
106 processed_featcol = add_required_geojson_properties(featcol_single_geom_type)
107 headers = {
108 "Content-Disposition": ("attachment; filename=geojson_withtags.geojson"),
109 "Content-Type": "application/media",
110 }
111 return Response(content=json.dumps(processed_featcol), headers=headers)
112
113 raise HTTPException(
114 status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
115 detail="Your geojson file is invalid.",
116 )

142@router.post("/create-entities-from-csv")
143async def create_entities_from_csv(
144 csv_file: UploadFile,
145 odk_project_id: int,
146 entity_name: str,
147 current_user: Annotated[AuthUser, Depends(login_required)],
148 odk_creds: ODKCentral = Depends(),
149):
150 """Upload a CSV file to create new ODK Entities in a project.
151
152 The Entity must already be defined on the server.
153 The CSV fields must match the Entity fields.
154 """
155 filename = Path(csv_file.filename)
156 file_ext = filename.suffix.lower()
157
158 if file_ext != ".csv":
159 raise HTTPException(
160 status_code=HTTPStatus.BAD_REQUEST, detail="Provide a valid .csv"
161 )
162
163 def parse_csv(csv_bytes):
164 parsed_data = []
165 csv_str = csv_bytes.decode("utf-8")
166 csv_reader = csv.DictReader(StringIO(csv_str))
167 for row in csv_reader:
168 parsed_data.append(dict(row))
169 return parsed_data
170
171 parsed_data = parse_csv(await csv_file.read())
172 entities_data_dict = {str(uuid4()): data for data in parsed_data}
173
174 async with central_deps.get_odk_dataset(odk_creds) as odk_central:
175 entities = await odk_central.createEntities(
176 odk_project_id,
177 entity_name,
178 entities_data_dict,
179 )
180
181 return entities

284@router.post("/send-test-osm-message")
285async def send_test_osm_message(
286 request: Request,
287 current_user: Annotated[AuthUser, Depends(login_required)],
288 # NOTE this is duplicated to access the 'deserialize_data' method
289 osm_auth: Annotated[Auth, Depends(init_osm_auth)],
290):
291 """Sends a test message to currently logged in OSM user."""
292 cookie_name = f"{settings.cookie_name}_osm"
293 log.debug(f"Extracting OSM token from cookie {cookie_name}")
294 serialised_osm_token = request.cookies.get(cookie_name)
295 if not serialised_osm_token:
296 return HTTPException(
297 status_code=HTTPStatus.UNAUTHORIZED,
298 detail="You must be logged in to your OpenStreetMap account.",
299 )
300
301 # NOTE to get this far, the user must be logged in using OSM
302 osm_token = osm_auth.deserialize_data(serialised_osm_token)
303 # NOTE message content must be in markdown format
304 message_content = dedent("""
305 # Heading 1
306
307 ## Heading 2
308
309 Hello there!
310
311 This is a text message in markdown format.
312
313 > Notes section
314 """)
315 # NOTE post body should contain either recipient or recipient_id
316 post_body = {
317 "recipient_id": 16289154,
318 # "recipient_id": current_user.id,
319 "title": "Test message from FMTM!",
320 "body": message_content,
321 }
322
323 email_url = f"{settings.OSM_URL}api/0.6/user/messages"
324 headers = {"Authorization": f"Bearer {osm_token}"}
325 log.debug(f"Sending message to user ({current_user.id}) via OSM API: {email_url}")
326 response = requests.post(email_url, headers=headers, data=post_body)
327
328 if response.status_code == 200:
329 log.info("Message sent successfully")
330 else:
331 msg = "Sending message via OSM failed"
332 log.error(f"{msg}: {response.text}")
333 return HTTPException(status_code=HTTPStatus.CONFLICT, detail=msg)
334
335 return Response(status_code=HTTPStatus.OK)
