110class TaskEvent(StrEnum, Enum):
111 """Task events via API.
112
113 `MAP` -- Set to *locked for mapping*, i.e. mapping in progress.
114 `FINISH` -- Set to *unlocked to validate*, i.e. is mapped.
115 `VALIDATE` -- Set to *locked for validation*, i.e. validation in progress.
116 `GOOD` -- Set the state to *unlocked done*.
117 `BAD` -- Set the state *unlocked to map* again, to be mapped once again.
118 `SPLIT` -- Set the state *unlocked done* then generate additional
119 subdivided task areas.
120 `MERGE` -- Set the state *unlocked done* then generate additional
121 merged task area.
122 `ASSIGN` -- For a requester user to assign a task to another user.
123 Set the state *locked for mapping* passing in the required user id.
124 Also notify the user they should map the area.
125 `COMMENT` -- Keep the state the same, but simply add a comment.
126 """
127
128 MAP = "MAP"
129 FINISH = "FINISH"
130 VALIDATE = "VALIDATE"
131 GOOD = "GOOD"
132 BAD = "BAD"
133 SPLIT = "SPLIT"
134 MERGE = "MERGE"
135 ASSIGN = "ASSIGN"
136 COMMENT = "COMMENT"
137 RESET = "RESET"
