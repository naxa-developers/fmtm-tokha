52def downloadThread(project_id: int, xforms: list, odk_credentials: dict, filters: dict = None):
53 """Download a list of submissions from ODK Central.
54
55 Args:
56 project_id (int): The ID of the project on ODK Central
57 xforms (list): A list of the XForms to down the submissions from
58 odk_credentials (dict): The authentication credentials for ODK Collect
59
60 Returns:
61 (list): The submissions in JSON format
62 """
63 timer = Timer(text="downloadThread() took {seconds:.0f}s")
64 timer.start()
65 data = list()
66 # log.debug(f"downloadThread() called! {len(xforms)} xforms")
67 for task in xforms:
68 form = OdkForm(odk_credentials["url"], odk_credentials["user"], odk_credentials["passwd"])
69 # submissions = form.getSubmissions(project_id, task, 0, False, True)
70 subs = form.listSubmissions(project_id, task, filters)
71 if not subs:
72 log.error(f"Failed to get submissions for project ({project_id}) task ({task})")
73 continue
74 # log.debug(f"There are {len(subs)} submissions for {task}")
75 if len(subs["value"]) > 0:
76 data += subs["value"]
77 # log.debug(f"There are {len(xforms)} Xforms, and {len(submissions)} submissions total")
78 timer.stop()
79 return data

83 def __init__(
84 self,
85 url: Optional[str] = None,
86 user: Optional[str] = None,
87 passwd: Optional[str] = None,
88 ):
89 """A Class for accessing an ODK Central server via it's REST API.
90
91 Args:
92 url (str): The URL of the ODK Central
93 user (str): The user's account name on ODK Central
94 passwd (str): The user's account password on ODK Central
95
96 Returns:
97 (OdkCentral): An instance of this class
98 """
99 if not url:
100 url = os.getenv("ODK_CENTRAL_URL", default=None)
101 self.url = url
102 if not user:
103 user = os.getenv("ODK_CENTRAL_USER", default=None)
104 self.user = user
105 if not passwd:
106 passwd = os.getenv("ODK_CENTRAL_PASSWD", default=None)
107 self.passwd = passwd
108 verify = os.getenv("ODK_CENTRAL_SECURE", default=True)
109 if type(verify) == str:
110 self.verify = verify.lower() in ("true", "1", "t")
111 else:
112 self.verify = verify
113 # Set cert bundle path for requests in environment
114 if self.verify:
115 os.environ["REQUESTS_CA_BUNDLE"] = "/etc/ssl/certs/ca-certificates.crt"
116 # These are settings used by ODK Collect
117 self.general = {
118 "form_update_mode": "match_exactly",
119 "autosend": "wifi_and_cellular",
120 }
121 # If there is a config file with authentication setting, use that
122 # so we don't have to supply this all the time. This is only used
123 # when odk_client is used, and no parameters are passed in.
124 if not self.url:
125 # log.debug("Configuring ODKCentral from file .odkcentral")
126 home = os.getenv("HOME")
127 config = ".odkcentral"
128 filespec = home + "/" + config
129 if os.path.exists(filespec):
130 file = open(filespec, "r")
131 for line in file:
132 # Support embedded comments
133 if line[0] == "#":
134 continue
135 # Read the config file for authentication settings
136 tmp = line.split("=")
137 if tmp[0] == "url":
138 self.url = tmp[1].strip("\n")
139 if tmp[0] == "user":
140 self.user = tmp[1].strip("\n")
141 if tmp[0] == "passwd":
142 self.passwd = tmp[1].strip("\n")
143 else:
144 log.warning(f"Authentication settings missing from {filespec}")
145 else:
146 log.debug(f"ODKCentral configuration parsed: {self.url}")
147 # Base URL for the REST API
148 self.version = "v1"
149 # log.debug(f"Using {self.version} API")
150 self.base = self.url + "/" + self.version + "/"
151
152 # Use a persistant connect, better for multiple requests
153 self.session = requests.Session()
154
155 # Authentication with session token
156 self.authenticate()
157
158 # These are just cached data from the queries
159 self.projects = dict()
160 self.users = list()

162 def authenticate(
163 self,
164 url: str = None,
165 user: str = None,
166 passwd: str = None,
167 ):
168 """Setup authenticate to an ODK Central server.
169
170 Args:
171 url (str): The URL of the ODK Central
172 user (str): The user's account name on ODK Central
173 passwd (str): The user's account password on ODK Central
174
175 Returns:
176 (requests.Response): A response from ODK Central after auth
177 """
178 if not self.url:
179 self.url = url
180 if not self.user:
181 self.user = user
182 if not self.passwd:
183 self.passwd = passwd
184 # Enable persistent connection, create a cookie for this session
185 self.session.headers.update({"accept": "odkcentral"})
186
187 # Get a session token
188 try:
189 response = self.session.post(
190 f"{self.base}sessions",
191 json={
192 "email": self.user,
193 "password": self.passwd,
194 },
195 )
196 except requests.exceptions.ConnectionError as request_error:
197 # URL does not exist
198 raise ConnectionError("Failed to connect to Central. Is the URL valid?") from request_error
199
200 if response.status_code == 401:
201 # Unauthorized, invalid credentials
202 raise ConnectionError("ODK credentials are invalid, or may have changed. Please update them.") from None
203 elif not response.ok:
204 # Handle other errors
205 response.raise_for_status()
206
207 self.session.headers.update({"Authorization": f"Bearer {response.json().get('token')}"})
208
209 # Connect to the server
210 return self.session.get(self.url, verify=self.verify)

212 def listProjects(self):
213 """Fetch a list of projects from an ODK Central server, and
214 store it as an indexed list.
215
216 Returns:
217 (list): A list of projects on a ODK Central server
218 """
219 log.info("Getting a list of projects from %s" % self.url)
220 url = f"{self.base}projects"
221 result = self.session.get(url, verify=self.verify)
222 projects = result.json()
223 for project in projects:
224 if isinstance(project, dict):
225 if project.get("id") is not None:
226 self.projects[project["id"]] = project
227 else:
228 log.info("No projects returned. Is this a first run?")
229 return projects

426 def getAllSubmissions(self, project_id: int, xforms: list = None, filters: dict = None):
427 """Fetch a list of submissions in a project on an ODK Central server.
428
429 Args:
430 project_id (int): The ID of the project on ODK Central
431 xforms (list): The list of XForms to get the submissions of
432
433 Returns:
434 (json): All of the submissions for all of the XForm in a project
435 """
436 # The number of threads is based on the CPU cores
437 info = get_cpu_info()
438 self.cores = info["count"]
439
440 timer = Timer(text="getAllSubmissions() took {seconds:.0f}s")
441 timer.start()
442 if not xforms:
443 xforms_data = self.listForms(project_id)
444 xforms = [d["xmlFormId"] for d in xforms_data]
445
446 chunk = round(len(xforms) / self.cores) if round(len(xforms) / self.cores) > 0 else 1
447 last_slice = len(xforms) if len(xforms) % chunk == 0 else len(xforms) - 1
448 cycle = range(0, (last_slice + chunk) + 1, chunk)
449 future = None
450 result = None
451 previous = 0
452 newdata = list()
453
454 # single threaded for easier debugging
455 # for current in cycle:
456 # if previous == current:
457 # continue
458 # result = downloadThread(project_id, xforms[previous:current])
459 # previous = current
460 # newdata += result
461
462 odk_credentials = {"url": self.url, "user": self.user, "passwd": self.passwd}
463
464 with concurrent.futures.ThreadPoolExecutor(max_workers=self.cores) as executor:
465 futures = list()
466 for current in cycle:
467 if previous == current:
468 continue
469 result = executor.submit(downloadThread, project_id, xforms[previous:current], odk_credentials, filters)
470 previous = current
471 futures.append(result)
472 for future in concurrent.futures.as_completed(futures):
473 log.debug("Waiting for thread to complete..")
474 data = future.result(timeout=10)
475 if len(data) > 0:
476 newdata += data
477 timer.stop()
478 return newdata

1016 def createForm(
1017 self,
1018 projectId: int,
1019 data: Union[str, Path, BytesIO],
1020 form_name: Optional[str] = None,
1021 publish: Optional[bool] = False,
1022 ) -> Optional[str]:
1023 """Create a new form on an ODK Central server.
1024
1025 - If no form_name is passed, the form name is generated by default in draft.
1026 If the publish param is also passed, then the form is published.
1027 - If form_name is passed, a new form is created from this in draft state.
1028 This copies across all attachments.
1029
1030 Note:
1031 The form name (xmlFormId) is generated from the id="" attribute
1032 immediately inside the <instance> tag of the XForm XML.
1033
1034 Args:
1035 projectId (int): The ID of the project on ODK Central
1036 form_name (str): The XForm to get the details of from ODK Central
1037 data (str, Path, BytesIO): The XForm file path, or BytesIO memory obj
1038 publish (bool): If the new form should be published.
1039 Only valid if form_name is not passed, i.e. a new form.
1040
1041 Returns:
1042 (str, Optional): The form name, else None if failure.
1043 """
1044 # BytesIO memory object
1045 if isinstance(data, BytesIO):
1046 self.xml = data.getvalue().decode("utf-8")
1047 # Filepath
1048 elif isinstance(data, str) or isinstance(data, Path):
1049 xml_path = Path(data)
1050 if not xml_path.exists():
1051 log.error(f"File does not exist on disk: {data}")
1052 return None
1053 # Read the XML or XLS file
1054 with open(xml_path, "rb") as xml_file:
1055 self.xml = xml_file.read()
1056 log.debug("Read %d bytes from %s" % (len(self.xml), data))
1057

1061 url = f"{self.base}projects/{projectId}/forms/{form_name}/draft?ignoreWarnings=true"
1062 else:
1063 # This is not a draft form, its an entirely new form (even if publish=false)
1064 log.debug("Creating new form, with name determined from form_id field")
1065 self.published = True if publish else False
1066 url = f"{self.base}projects/{projectId}/forms?ignoreWarnings=true&{'publish=true' if publish else ''}"
1067
1068 result = self.session.post(
1069 url, data=self.xml, headers=dict({"Content-Type": "application/xml"}, **self.session.headers), verify=self.verify
1070 )
1071

1080 if last_full_stop_index != -1 and last_comma_index != -1:
1081 # Extract xmlFormId from error msg
1082 xmlFormId = msg[last_comma_index + 1 : last_full_stop_index].strip()
1083 return xmlFormId
1084 else:
1085 log.warning("Unable to extract xmlFormId from error message")
1086 return None
1087 else:
1088 log.error(f"Couldn't create {form_name} on Central: {msg}")
1089 return None
1090 except json.decoder.JSONDecodeError:
1091 log.error(f"Couldn't create {form_name} on Central: Error decoding JSON response")
1092 return None
1093
1094 try:
1095 # Log response to terminal
1096 json_data = result.json()
1097 except json.decoder.JSONDecodeError:
1098 log.error("Could not parse response json during form creation")
1099 return None

1102 # FIXME: should update self.forms with the new form
1103

1106 return form_name
