421@router.get("/convert-fgb-to-geojson")
422async def convert_fgb_to_geojson(
423 url: str,
424 db: Annotated[Connection, Depends(db_conn)],
425 current_user: Annotated[AuthUser, Depends(login_required)],
426):
427 """Convert flatgeobuf to GeoJSON format, extracting GeometryCollection.
428
429 Helper endpoint to test data extracts during project creation.
430 Required as the flatgeobuf files wrapped in GeometryCollection
431 cannot be read in QGIS or other tools.
432
433 Args:
434 url (str): URL to the flatgeobuf file.
435 db (Connection): The database connection.
436 current_user (AuthUser): Check if user is logged in.
437
438 Returns:
439 Response: The HTTP response object containing the downloaded file.
440 """
441 with requests.get(url) as response:
442 if not response.ok:
443 raise HTTPException(
444 status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
445 detail="Download failed for data extract",
446 )
447 data_extract_geojson = await flatgeobuf_to_featcol(db, response.content)
448
449 if not data_extract_geojson:
450 raise HTTPException(
451 status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
452 detail=("Failed to convert flatgeobuf --> geojson"),
453 )
454
455 headers = {
456 "Content-Disposition": ("attachment; filename=fmtm_data_extract.geojson"),
457 "Content-Type": "application/media",
458 }
459
460 return Response(content=json.dumps(data_extract_geojson), headers=headers)

498@router.post("/task-split")
499async def task_split(
500 # NOTE we do not set any roles on this endpoint yet
501 # FIXME once sub project creation implemented, this should be manager only
502 current_user: Annotated[AuthUser, Depends(login_required)],
503 project_geojson: UploadFile = File(...),
504 extract_geojson: Optional[UploadFile] = File(None),
505 no_of_buildings: int = Form(50),
506):
507 """Split a task into subtasks.
508
509 NOTE we pass a connection
510
511 Args:
512 current_user (AuthUser): the currently logged in user.
513 project_geojson (UploadFile): The geojson (AOI) to split.
514 extract_geojson (UploadFile, optional): Custom data extract geojson
515 containing osm features (should be a FeatureCollection).
516 If not included, an extract is generated automatically.
517 no_of_buildings (int, optional): The number of buildings per subtask.
518 Defaults to 50.
519
520 Returns:
521 The result of splitting the task into subtasks.
522 """
523 boundary_featcol = parse_geojson_file_to_featcol(await project_geojson.read())
524 merged_boundary = merge_polygons(boundary_featcol, False)
525 # Validatiing Coordinate Reference Systems
526 await check_crs(merged_boundary)
527
528 # read data extract
529 parsed_extract = None
530 if extract_geojson:
531 parsed_extract = parse_geojson_file_to_featcol(await extract_geojson.read())
532 if parsed_extract:
533 await check_crs(parsed_extract)
534 else:
535 log.warning("Parsed geojson file contained no geometries")
536
537 log.debug("STARTED task splitting using provided boundary and data extract")
538 # NOTE here we pass the connection string and allow fmtm-splitter to
539 # a use psycopg2 connection (not async)
540 features = await run_in_threadpool(
541 lambda: split_by_sql(
542 merged_boundary,
543 settings.FMTM_DB_URL.unicode_string(),
544 num_buildings=no_of_buildings,
545 osm_extract=parsed_extract,
546 )
547 )
548 log.debug("COMPLETE task splitting")
549 return features

594@router.post("/preview-split-by-square", response_model=FeatureCollection)
595async def preview_split_by_square(
596 # NOTE we do not set any roles on this endpoint yet
597 # FIXME once sub project creation implemented, this should be manager only
598 current_user: Annotated[AuthUser, Depends(login_required)],
599 project_geojson: UploadFile = File(...),
600 extract_geojson: Optional[UploadFile] = File(None),
601 dimension_meters: int = Form(100),
602):
603 """Preview splitting by square.
604
605 TODO update to use a response_model
606 """
607 # Validating for .geojson File.
608 file_name = os.path.splitext(project_geojson.filename)
609 file_ext = file_name[1]
610 allowed_extensions = [".geojson", ".json"]
611 if file_ext not in allowed_extensions:
612 raise HTTPException(
613 HTTPStatus.BAD_REQUEST,
614 detail="Provide a valid .geojson file",
615 )
616
617 # read entire file
618 boundary_featcol = parse_geojson_file_to_featcol(await project_geojson.read())
619
620 # Validatiing Coordinate Reference System
621 await check_crs(boundary_featcol)
622 parsed_extract = None
623 if extract_geojson:
624 parsed_extract = parse_geojson_file_to_featcol(await extract_geojson.read())
625 if parsed_extract:
626 await check_crs(parsed_extract)
627 else:
628 log.warning("Parsed geojson file contained no geometries")
629
630 if len(boundary_featcol["features"]) > 0:
631 boundary_featcol = merge_polygons(boundary_featcol)
632
633 return split_by_square(
634 boundary_featcol,
635 settings.FMTM_DB_URL.unicode_string(),
636 meters=dimension_meters,
637 osm_extract=parsed_extract,
638 )
