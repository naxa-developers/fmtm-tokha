95@router.get("/download")
96async def download_submission(
97 project_user: Annotated[ProjectUserDict, Depends(project_contributors)],
98 file_type: SubmissionDownloadType,
99 submitted_date_range: Optional[str] = Query(
100 None,
101 title="Submitted Date Range",
102 description="Date range in format (e.g., 'YYYY-MM-DD,YYYY-MM-DD')",
103 ),
104):
105 """Download the submissions for a given project.
106
107 Returns a JSON, CSV, or GeoJSON file.
108 """
109 project = project_user.get("project")
110 filters = {}
111
112 if submitted_date_range:
113 start_date, end_date = submitted_date_range.split(",")
114 filters = {
115 "$filter": (
116 f"__system/submissionDate ge {start_date}T00:00:00+00:00 "
117 f"and __system/submissionDate le {end_date}T23:59:59.999+00:00"
118 )
119 }
120
121 if file_type == SubmissionDownloadType.JSON:
122 return await submission_crud.download_submission_in_json(project, filters)
123
124 elif file_type == SubmissionDownloadType.CSV:
125 file_content = await submission_crud.gather_all_submission_csvs(
126 project, filters
127 )
128 headers = {"Content-Disposition": f"attachment; filename={project.slug}.zip"}
129 return Response(file_content, headers=headers)
130
131 # Else is GeoJSON download
132 data = await submission_crud.get_submission_by_project(project, filters)
133 submission_json = data.get("value", [])
134
135 submission_geojson = await central_crud.convert_odk_submission_json_to_geojson(
136 submission_json
137 )
138 submission_data = BytesIO(json.dumps(submission_geojson).encode("utf-8"))
139
140 headers = {"Content-Disposition": f"attachment; filename={project.slug}.geojson"}
141 return Response(submission_data.getvalue(), headers=headers)

297@router.get("/submission-table")
298async def submission_table(
299 project_user: Annotated[ProjectUserDict, Depends(mapper)],
300 page: int = Query(1, ge=1),
301 results_per_page: int = Query(13, le=100),
302 task_id: Optional[int] = None,
303 submitted_by: Optional[str] = None,
304 review_state: Optional[str] = None,
305 submitted_date_range: Optional[str] = Query(
306 None,
307 title="Submitted Date Range",
308 description="Date range in format (e.g., 'YYYY-MM-DD,YYYY-MM-DD')",
309 ),
310):
311 """This api returns the submission table of a project.
312
313 Returns:
314 dict: Paginated submissions with `results` and `pagination` keys.
315 """
316 project = project_user.get("project")
317 filters = {
318 "$count": True,
319 "$wkt": True,
320 }
321
322 if submitted_date_range:
323 start_date, end_date = submitted_date_range.split(",")
324 filters["$filter"] = (
325 "__system/submissionDate ge {}T00:00:00+00:00 "
326 "and __system/submissionDate le {}T23:59:59.999+00:00"
327 ).format(start_date, end_date)
328
329 if review_state and review_state != "received":
330 review_filter = f"__system/reviewState eq '{review_state}'"
331 filters["$filter"] = (
332 f"{filters['$filter']} and {review_filter}"
333 if "$filter" in filters
334 else review_filter
335 )
336
337 data = await submission_crud.get_submission_by_project(project, filters)
338 total_count = data.get("@odata.count", 0)
339 submissions = data.get("value", [])
340 if review_state == "received":
341 submissions = [
342 sub for sub in submissions if sub["__system"].get("reviewState") is None
343 ]
344 total_count = len(submissions)
345
346 if task_id:
347 submissions = [sub for sub in submissions if sub.get("task_id") == str(task_id)]
348
349 if submitted_by:
350 submissions = [
351 sub for sub in submissions if sub.get("username") == submitted_by
352 ]
353
354 start_index = (page - 1) * results_per_page
355 end_index = start_index + results_per_page
356 paginated_submissions = submissions[start_index:end_index]
357
358 pagination = await project_crud.get_pagination(
359 page, len(submissions), results_per_page, total_count
360 )
361 response = submission_schemas.PaginatedSubmissions(
362 results=paginated_submissions,
363 pagination=submission_schemas.PaginationInfo(**pagination.model_dump()),
364 )
365
366 return response

437@router.get("/conflate-submission-geojson")
438async def conflate_geojson(
439 db_task: Annotated[DbTask, Depends(get_task)],
440 project_user: Annotated[ProjectUserDict, Depends(mapper)],
441 remove_conflated: Annotated[
442 bool,
443 Query(
444 description="Removes geometries not overlapping with OSM data",
445 ),
446 ] = False,
447):
448 """Conflates the input GeoJSON with OpenStreetMap data.
449
450 Returns:
451 str: Updated GeoJSON string with conflated features.
452 """
453 try:
454 project = project_user.get("project")
455 task_geojson = geojson.dumps(db_task.outline, indent=2)
456
457 data = await submission_crud.get_submission_by_project(project, {})
458 submission_json = data.get("value", [])
459 task_submission = [
460 sub for sub in submission_json if sub["task_id"] == str(db_task.id)
461 ]
462
463 if not task_submission:
464 return JSONResponse(
465 status_code=HTTPStatus.NOT_FOUND,
466 content=f"No Submissions found within the task {db_task.id}",
467 )
468
469 submission_geojson = await central_crud.convert_odk_submission_json_to_geojson(
470 task_submission
471 )
472 osm_category = project.osm_category
473 input_features = submission_geojson["features"]
474
475 osm_features = postgis_utils.get_osm_geometries(osm_category, task_geojson)
476 conflated_features = postgis_utils.conflate_features(
477 input_features, osm_features.get("features", []), remove_conflated
478 )
479 submission_geojson["features"] = conflated_features
480
481 return submission_geojson
482 except HTTPException as e:
483 raise e
484 except Exception as e:
485 raise HTTPException(
486 HTTPStatus.INTERNAL_SERVER_ERROR,
487 detail=f"Failed to process conflation: {str(e)}",
488 ) from e
