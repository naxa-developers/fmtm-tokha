56class OtelSettings(BaseSettings):
57 """Inherited OpenTelemetry specific settings (monitoring).
58
59 These mostly set environment variables set by the OTEL SDK.
60 """
61
62 FMTM_DOMAIN: Optional[str] = Field(exclude=True)
63 LOG_LEVEL: Optional[str] = Field(exclude=True)

66 @computed_field
67 @property
68 def otel_log_level(self) -> Optional[str]:
69 """Set OpenTelemetry log level."""
70 log_level = "info"
71 if self.LOG_LEVEL:
72 log_level = self.LOG_LEVEL.lower()
73 # NOTE setting to DEBUG makes very verbose for every library
74 # os.environ["OTEL_LOG_LEVEL"] = log_level
75 os.environ["OTEL_LOG_LEVEL"] = "info"
76 return log_level
77
78 @computed_field
79 @property
80 def otel_service_name(self) -> Optional[HttpUrlStr]:
81 """Set OpenTelemetry service name for traces."""
82 service_name = "unknown"
83 if self.FMTM_DOMAIN:
84 # Return domain with underscores
85 service_name = self.FMTM_DOMAIN.replace(".", "_")
86 # Export to environment for OTEL instrumentation
87 os.environ["OTEL_SERVICE_NAME"] = service_name
88 return service_name
89
90 @computed_field
91 @property
92 def otel_python_excluded_urls(self) -> Optional[str]:
93 """Set excluded URLs for Python instrumentation."""
94 endpoints = "__lbheartbeat__,docs,openapi.json"
95 os.environ["OTEL_PYTHON_EXCLUDED_URLS"] = endpoints
96 # Add extra endpoints ignored by for requests
97 # NOTE we add ODK Central session auth endpoint here
98 if self.ODK_CENTRAL_URL:
99 os.environ["OTEL_PYTHON_REQUESTS_EXCLUDED_URLS"] = (
100 f"{endpoints}{self.ODK_CENTRAL_URL}/v1/sessions"
101 )
102 return endpoints
103
104 @computed_field
105 @property
106 def otel_python_log_correlation(self) -> Optional[str]:
107 """Set log correlation for OpenTelemetry Python spans."""
108 value = "true"
109 os.environ["OTEL_PYTHON_LOG_CORRELATION"] = value
110 return value

169 @field_validator("EXTRA_CORS_ORIGINS", mode="before")
170 @classmethod
171 def assemble_cors_origins(
172 cls,
173 extra_origins: Optional[Union[str, list[str]]],
174 info: ValidationInfo,
175 ) -> list[str]:
176 """Build and validate CORS origins list."""
177 # Initialize default origins
178 default_origins = ["https://xlsforms.fmtm.dev"]
179
180 # Handle localhost/testing scenario
181 domain = info.data.get("FMTM_DOMAIN", "fmtm.localhost")
182 dev_port = info.data.get("FMTM_DEV_PORT")
183 # NOTE fmtm.dev.test is used as the Playwright test domain
184 if "localhost" in domain or "fmtm.dev.test" in domain:
185 local_server_port = (
186 f":{dev_port}"
187 if dev_port and dev_port.lower() not in ("0", "no", "false")
188 else ""
189 )
190 default_origins.append(f"http://{domain}{local_server_port}")
191 else:
192 # Add the main FMTM domain
193 default_origins.append(f"https://{domain}")
194
195 # Process `extra_origins` if provided
196 if isinstance(extra_origins, str):
197 # Split by comma and strip whitespace
198 extra_origins_list = [
199 i.strip() for i in extra_origins.split(",") if i.strip()
200 ]
201 default_origins.extend(extra_origins_list)
202 elif isinstance(extra_origins, list):
203 default_origins.extend(extra_origins)
204
205 # Ensure uniqueness and return (remove dups)
206 return list(dict.fromkeys(default_origins))

254 def configure_s3_download_root(cls, v: Optional[str], info: ValidationInfo) -> str:
255 """Set S3_DOWNLOAD_ROOT for S3 downloads.
256
257 This is required, when we use a containerised S3 service.
258 The S3_ENDPOINT is a docker compose service name and not
259 resolvable outside of the stack.
260
261 S3_DOWNLOAD_ROOT is equal to S3_ENDPOINT if a public S3 instance

263 """
264 # If set manually, pass through
265 if v and isinstance(v, str):
266 return v

269 s3_endpoint = info.data.get("S3_ENDPOINT")

271 return s3_endpoint
272
273 # Containerised S3
274 else:
275 fmtm_domain = info.data.get("FMTM_DOMAIN")
276 # Local dev
277 # NOTE for automated tests, this is overridden manually
278 if info.data.get("DEBUG"):
279 dev_port = info.data.get("FMTM_DEV_PORT")
280 return f"http://s3.{fmtm_domain}:{dev_port}"
281 return f"https://s3.{fmtm_domain}"

323@lru_cache
324def get_cipher_suite():
325 """Cache cypher suite."""
326 # Fernet is used by cryptography as a simple and effective default
327 # it enforces a 32 char secret.
328 #
329 # In the future we could migrate this to HS384 encryption, which we also
330 # use for our JWT signing. Ideally this needs 48 characters, but for now
331 # we are stuck at 32 char to maintain support with Fernet (reuse the same key).
332 #
333 # However this would require a migration for all existing instances of FMTM.
334 return Fernet(settings.ENCRYPTION_KEY.get_secret_value())

337def encrypt_value(password: Union[str, HttpUrlStr]) -> str:
338 """Encrypt value before going to the DB."""
339 cipher_suite = get_cipher_suite()
340 encrypted_password = cipher_suite.encrypt(password.encode("utf-8"))
341 return base64.b64encode(encrypted_password).decode("utf-8")
342
343
344def decrypt_value(db_password: str) -> str:
345 """Decrypt the database value."""
346 cipher_suite = get_cipher_suite()
347 encrypted_password = base64.b64decode(db_password.encode("utf-8"))
348 decrypted_password = cipher_suite.decrypt(encrypted_password)
349 return decrypted_password.decode("utf-8")
