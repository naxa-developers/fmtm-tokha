58@router.get("/api-key")
59async def get_api_key(
60 current_user: Annotated[DbUser, Depends(super_admin)],
61 db: Annotated[Connection, Depends(db_conn)],
62):
63 """Generate and return a new API key.
64
65 This can only be accessed once, and is regenerated on
66 each call to this endpoint.
67
68 Be sure to store it someplace safe, like a password manager.
69
70 NOTE currently requires super admin permission.
71 """
72 try:
73 api_token = await generate_api_token(db, current_user.id)
74 except ValueError as e:
75 raise HTTPException(
76 status_code=HTTPStatus.INTERNAL_SERVER_ERROR,
77 detail=str(e),
78 ) from e
79 return JSONResponse(
80 status_code=HTTPStatus.OK,
81 content={"api_token": api_token},
82 )

107 elif odk_event.type == OdkWebhookEvents.REVIEW_SUBMISSION:
108 # update entity status in odk
109 # await update_entity_status_in_odk(db, odk_event)

113
114 elif odk_event.type == OdkWebhookEvents.NEW_SUBMISSION:
115 # unsupported for now
116 log.debug("The handling of new submissions via webhook is not implemented yet.")
117
118 else:
119 msg = (
120 f"Webhook was called for an unsupported event type ({odk_event.type.value})"
121 )
