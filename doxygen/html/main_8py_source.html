158class InterceptHandler(logging.Handler):
159 """Intercept python standard lib logging."""
160
161 def emit(self, record):
162 """Retrieve context where the logging call occurred.
163
164 This happens to be in the 6th frame upward.
165 """
166 logger_opt = log.opt(depth=6, exception=record.exc_info)
167 logger_opt.log(logging.getLevelName(record.levelno), record.getMessage())

170def get_logger():
171 """Override FastAPI logger with custom loguru."""
172 # Hook all other loggers into ours
173 logger_name_list = [name for name in logging.root.manager.loggerDict]
174 for logger_name in logger_name_list:
175 logging.getLogger(logger_name).setLevel(settings.LOG_LEVEL)
176 logging.getLogger(logger_name).handlers = []
177 if logger_name == "urllib3":
178 # Don't hook urllib3, called on each OTEL trace
179 continue
180 if "." not in logger_name:
181 logging.getLogger(logger_name).addHandler(InterceptHandler())
182
183 log.remove()
184 log.add(
185 sys.stderr,
186 level=settings.LOG_LEVEL,
187 format=(
188 "{time:YYYY-MM-DD HH:mm:ss.SSS} | {level: <8} "
189 "| {name}:{function}:{line} | {message}"
190 ),
191 enqueue=True, # Run async / non-blocking
192 colorize=True,
193 backtrace=True, # More detailed tracebacks
194 catch=True, # Prevent app crashes
195 )
196
197 # Only log to file in production
198 if not settings.DEBUG:
199 log.add(
200 "/opt/logs/fmtm.json",
201 level=settings.LOG_LEVEL,
202 enqueue=True,
203 serialize=True, # JSON format
204 rotation="00:00", # New file at midnight
205 retention="10 days",
206 # format=log_json_format, # JSON format func
