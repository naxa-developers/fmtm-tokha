132@router.post("/approve", response_model=OrganisationOut)
133async def approve_organisation(
134 request: Request,
135 org_id: int,
136 background_tasks: BackgroundTasks,
137 db: Annotated[Connection, Depends(db_conn)],
138 current_user: Annotated[AuthUser, Depends(super_admin)],
139 osm_auth=Depends(init_osm_auth),
140):
141 """Approve the organisation request made by the user.
142
143 The logged in user must be super admin to perform this action.
144
145 A background task notifies the organisation creator.
146 """
147 log.info(f"Approving organisation ({org_id}).")
148 approved_org = await DbOrganisation.update(
149 db,
150 org_id,
151 OrganisationUpdate(approved=True),
152 )
153 # Set organisation requester as organisation manager
154 if approved_org.created_by:
155 await DbOrganisationManagers.create(
156 db, approved_org.id, approved_org.created_by
157 )
158
159 log.info(f"Approved organisation ({org_id}).")
160 background_tasks.add_task(
161 organisation_crud.send_approval_message,
162 request=request,
163 creator_id=approved_org.created_by,
164 organisation_name=approved_org.name,
165 osm_auth=osm_auth,
166 )
167
168 return approved_org
